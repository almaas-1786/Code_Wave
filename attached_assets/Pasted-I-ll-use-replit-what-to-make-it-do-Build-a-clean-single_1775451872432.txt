🧠 6. MESSAGE ANALYZER
const urgencyWords = ["urgent", "immediately", "now"];
const emotionWords = ["help", "please", "trust"];
const authorityWords = ["bank", "account", "kyc"];

function analyzeMessage(text) {
  let score = 0;
  let signals = [];

  let lower = text.toLowerCase();

  urgencyWords.forEach(w => {
    if (lower.includes(w)) {
      score += 20;
      signals.push("Urgency");
    }
  });

  emotionWords.forEach(w => {
    if (lower.includes(w)) {
      score += 15;
      signals.push("Emotional Pressure");
    }
  });

  authorityWords.forEach(w => {
    if (lower.includes(w)) {
      score += 20;
      signals.push("Authority");
    }
  });

  return { score, signals };
}

💸 7. PAYMENT ANALYSIS
let knownRecipients = ["mom@upi", "friend@upi"];

function analyzePayment(data) {
  let score = 0;
  let signals = [];

  if (data.amount > 5000) {
    score += 30;
    signals.push("High Amount");
  }

  if (!knownRecipients.includes(data.upi)) {
    score += 25;
    signals.push("New Recipient");
  }

  if (data.note.toLowerCase().includes("urgent")) {
    score += 20;
    signals.push("Urgent Request");
  }

  return { score, signals };
}
🔗 COMBINE BOTH
function getFinalRisk(msg, pay) {
  let total = msg.score + pay.score;
  let signals = [...msg.signals, ...pay.signals];

  let level = "Low";
  if (total > 80) level = "High";
  else if (total > 50) level = "Medium";

  return { total, signals, level };
}

function handlePayment() {
  let data = {
    name: document.getElementById("name").value,
    upi: document.getElementById("upi").value,
    amount: parseInt(document.getElementById("amount").value),
    note: document.getElementById("note").value
  };

  let paymentResult = analyzePayment(data);
  let final = getFinalRisk(lastMessageResult, paymentResult);

  showResult(final);
}
